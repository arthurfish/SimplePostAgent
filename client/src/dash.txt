import { useLiveRequests } from "../../hooks/useLiveQueries";
import { Spinner } from "../ui/Spinner";
import { Link } from "react-router-dom";

// Refresh icon for "In Progress" tasks
const RefreshIcon = () => (
    <svg xmlns="http://www.w3.org/2000/svg" width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2.5" strokeLinecap="round" strokeLinejoin="round" className="animate-spin" style={{ animationDuration: '2s' }}>
        <path d="M21 12a9 9 0 1 1-6.219-8.56"/>
    </svg>
);

const getStatusStyles = (status: string) => {
    switch (status.toLowerCase()) {
        case 'completed':
            return {
                cardClass: 'bg-gradient-to-br from-[#4CD0C1] to-[#39A497]',
                accentClass: 'bg-[#297C72]',
            };
        case 'in progress':
            return {
                cardClass: 'bg-gradient-to-br from-[#DD6A6E] to-[#C95358]',
                accentClass: 'bg-[#984545]',
            };
        case 'pending':
            return {
                cardClass: 'bg-gradient-to-br from-[#807EE8] to-[#6A66D9]',
                accentClass: 'bg-[#4A47A5]',
            };
        default:
            return {
                cardClass: 'bg-gray-400',
                accentClass: 'bg-gray-600',
            };
    }
};

const TaskCard = ({ req }: { req: any }) => {
    const { cardClass, accentClass } = getStatusStyles(req.status);
    const formattedStatus = req.status.replace(/_/g, ' ').replace(/\b\w/g, l => l.toUpperCase());

    return (
        <Link to={`/request/${req.id}`} className="relative block group">
            {/* Decorative accent bars */}
            {req.status === 'completed' && <div className={`absolute -top-1.5 left-6 h-2 w-20 ${accentClass} rounded-md`}></div>}
            {req.status === 'in progress' && (
                <>
                    <div className={`absolute top-6 -left-1.5 h-16 w-2 ${accentClass} rounded-md`}></div>
                    <div className={`absolute bottom-6 -right-1.5 h-16 w-2 ${accentClass} rounded-md`}></div>
                </>
            )}
            {req.status === 'pending' && (
                <>
                    <div className={`absolute top-6 -left-1.5 h-12 w-2 ${accentClass} rounded-md`}></div>
                    <div className={`absolute bottom-6 -right-1.5 h-12 w-2 ${accentClass} rounded-md`}></div>
                </>
            )}

            <div className={`relative p-6 rounded-xl text-white shadow-lg transition-transform transform group-hover:scale-105 duration-200 ease-in-out ${cardClass}`}>
                {req.status === 'in progress' && (
                    <div className="absolute -top-4 -right-4 bg-[#F2A855] text-white p-2.5 rounded-lg shadow-md">
                        <RefreshIcon />
                    </div>
                )}
                <h3 className="text-xl font-bold mb-1">Task #{String(req.id).padStart(3, '0')}</h3>
                <p className="text-base font-light opacity-90">Status: {formattedStatus}</p>
            </div>
        </Link>
    );
};

export const RequestList = () => {
    const { data: requests, isLoading, error } = useLiveRequests();

    if (isLoading) {
        return <div className="flex justify-center p-8"><Spinner /></div>;
    }

    if (error) {
        return <p className="text-red-500">Failed to load requests. Please try again later.</p>
    }

    if (requests.length === 0) {
        return <p className="text-center text-gray-500 py-8">No tasks found. Launch one to get started!</p>;
    }

    return (
        <div className="grid gap-x-8 gap-y-12 md:grid-cols-2">
            {requests.map((req) => (
                <TaskCard req={req} key={req.id} />
            ))}
            {/* Static placeholder card to match the design's composition */}
            <div className="relative">
                <div className="absolute top-6 -left-1.5 h-12 w-2 bg-slate-400/80 rounded-md"></div>
                <div className="absolute bottom-6 -right-1.5 h-12 w-2 bg-slate-400/80 rounded-md"></div>
                <div className="relative p-6 rounded-xl bg-[#C8C5E5] h-full min-h-[124px] flex flex-col justify-end shadow-lg">
                    <div className="w-full bg-white/30 rounded-full h-2.5">
                        <div className="bg-white/50 h-2.5 rounded-full" style={{ width: '45%' }}></div>
                    </div>
                </div>
            </div>
        </div>
    );
};// src/components/requests/NewRequestForm.tsx
import { useState } from "react";
import { Effect } from "effect";
import { Spinner } from "../ui/Spinner";
import { useApiService } from "../AppProvider.tsx";

// Paper plane icon for the launch button, adjusted to match the design
const PaperPlaneIcon = () => (
    <svg xmlns="http://www.w3.org/2000/svg" width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round">
        <path d="m22 2-7 20-4-9-9-4Z"/><path d="M22 2 11 13"/>
    </svg>
);

export const NewRequestForm = () => {
    const [input, setInput] = useState("");
    const [isLoading, setIsLoading] = useState(false);
    const [error, setError] = useState<string | null>(null);
    const apiService = useApiService();

    const handleSubmit = (e: React.FormEvent) => {
        e.preventDefault();
        if (!input.trim() || isLoading) return;

        setIsLoading(true);
        setError(null);

        const program = apiService.createRequest(input).pipe(
            Effect.tap(() => Effect.sync(() => setInput(""))),
            Effect.catchAll((e) => Effect.sync(() => setError(e.message))),
            Effect.ensuring(Effect.sync(() => setIsLoading(false)))
        );

        Effect.runFork(program);
    };

    return (
        <div className="relative">
            <form onSubmit={handleSubmit} className="flex items-center gap-3">
                <input
                    type="text"
                    placeholder="Enter your request..."
                    value={input}
                    onChange={(e) => setInput(e.target.value)}
                    disabled={isLoading}
                    className="flex-grow p-4 rounded-md border-none bg-[#F7F7F7] placeholder:text-[#C1C1C1] focus:outline-none"
                />
                <button
                    type="submit"
                    disabled={isLoading}
                    className="flex items-center justify-center gap-2 px-6 py-4 bg-[linear-gradient(to_bottom_right,#F07F79,#D8605A)] text-white font-semibold rounded-md hover:opacity-90 transition-all active:scale-95 disabled:opacity-50 disabled:cursor-not-allowed"
                >
                    {isLoading ? <Spinner size="sm" /> : (
                        <>
                            <PaperPlaneIcon />
                            Launch
                        </>
                    )}
                </button>
            </form>
            {error && <p className="text-red-500 text-sm absolute -bottom-6 left-2">{error}</p>}
        </div>
    );
};
